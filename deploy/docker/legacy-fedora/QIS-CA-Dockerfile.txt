# QIS Cache server
#
# Build notes:
#   Standard fedora/memcached image but with a configurable cache size.
#   The default environment variables assume a memory limit of about 1GB.
#
# Files required in current directory:
#   None
# 
# Build/Environment variables:
#   CACHESIZE (Optional - to override the runtime size of the cache)
#
FROM fedora/memcached
MAINTAINER Matt Fozard <matt@quru.com>

# Set the default cache size in MB (override this with docker run --env CACHESIZE=n)
ENV CACHESIZE 512

# Overrides fedora/memcached to run as memcached user and set the memory limit
CMD /usr/bin/memcached -u memcached -m $CACHESIZE
